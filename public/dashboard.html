// routes/whatsappRoutes.js const express = require('express'); const router = express.Router(); const fs = require('fs'); const qrcode = require('qrcode-terminal'); const { Client, LocalAuth } = require('whatsapp-web.js'); const db = require('../db'); const learningHandler = require('../src/handlers/learningHandler'); // Variable global para almacenar el QR actual let currentQR = null; // Crear una instancia √∫nica del cliente let client = new Client({ authStrategy: new LocalAuth(), puppeteer: { headless: true, args: ['--no-sandbox', '--disable-setuid-sandbox'] } }); /* ============================== EVENTOS DEL CLIENTE ============================== */ // Cuando se genera un QR client.on('qr', (qr) => { console.log('üì± Se gener√≥ un nuevo QR (whatsapp-web.js)'); // Mostrarlo en la consola de forma textual qrcode.generate(qr, { small: true }); // Guardarlo en la variable global currentQR = qr; }); // Cuando el cliente est√° listo client.on('ready', () => { console.log('ü§ñ Bot de WhatsApp listo y en modo aprendizaje'); // Si est√° conectado, ya no necesitamos el QR currentQR = null; }); // Cuando llega un mensaje entrante client.on('message', async (msg) => { const remitente = msg.from; const mensaje = msg.body; const fecha = new Date().toISOString(); // Guardar en DB (ejemplo con fromMe=0 para el usuario) db.run(` INSERT INTO mensajes (remitente, mensaje, fecha, fromMe) VALUES (?, ?, ?, 0) `, [remitente, mensaje, fecha], (err) => { if (err) { console.error('‚ùå Error al guardar mensaje de usuario:', err.message); } else { console.log(`üíæ Mensaje de usuario guardado: "${mensaje}" de ${remitente}`); } }); // L√≥gica de aprendizaje (opcional) await learningHandler.processMessage(msg); }); // Cuando se crea un mensaje (incluye los que env√≠a el bot) client.on('message_create', async (msg) => { // Solo guardamos si lo env√≠a el bot if (msg.fromMe) { const remitente = msg.to; const mensaje = msg.body; const fecha = new Date().toISOString(); db.run(` INSERT INTO mensajes (remitente, mensaje, fecha, fromMe) VALUES (?, ?, ?, 1) `, [remitente, mensaje, fecha], (err) => { if (err) { console.error('‚ùå Error al guardar mensaje del bot:', err.message); } else { console.log(`üíæ Mensaje del bot guardado: "${mensaje}" para ${remitente}`); } }); } }); /* ============================== INICIALIZAR CLIENTE ============================== */ client.initialize().then(() => { console.log('‚úÖ Cliente de WhatsApp inicializado'); }).catch(err => { console.error('Error al inicializar cliente:', err); }); /* ============================== RUTAS ============================== */ // 1) Endpoint /qr-status para que el frontend sepa si est√° conectado o no router.get('/qr-status', async (req, res) => { try { const state = await client.getState(); // Puede ser CONNECTED, DISCONNECTED, etc. if (state === 'CONNECTED') { return res.json({ state: 'CONNECTED', qr: null }); } // Si no est√° conectado, devolvemos el QR // Nota: si se destruy√≥ el client o pupPage es null, devu√©lvelo como 'DISCONNECTED' if (!client.pupPage) { return res.json({ state: 'DISCONNECTED', qr: null }); } // De lo contrario, asumimos que est√° "CONNECTING" y devolvemos el currentQR return res.json({ state: 'CONNECTING', qr: currentQR }); } catch (err) { console.error('Error al obtener estado:', err); return res.json({ state: 'DISCONNECTED', qr: null }); } }); // 2) Ruta /generate-qr que sirve el HTML con el script que hace polling router.get('/generate-qr', async (req, res) => { try { // Revisamos si ya est√° conectado const state = await client.getState(); if (state === 'CONNECTED') { return res.send(`

‚úÖ Bot ya est√° conectado

Volver al panel

`); } // Si no est√° conectado, devolvemos la p√°gina que hace polling a /qr-status res.send(` Generando c√≥digo QR

Generando c√≥digo QR...

Conectando...

`); } catch (err) { console.error('Error al generar QR:', err); res.status(500).send('Error al generar QR. Intenta nuevamente.'); } }); // 3) Ruta para cerrar sesi√≥n (logout) router.get('/logout', async (req, res) => { try { await client.logout(); console.log('‚úÖ Sesi√≥n de WhatsApp cerrada correctamente'); // Eliminar carpeta de autenticaci√≥n si deseas const sessionPath = './.wwebjs_auth'; if (fs.existsSync(sessionPath)) { fs.rmSync(sessionPath, { recursive: true, force: true }); console.log('üóë Carpeta de sesi√≥n eliminada'); } // Destruir el cliente await client.destroy(); console.log('üîÑ Cliente destruido.'); // Crear uno nuevo client = new Client({ authStrategy: new LocalAuth(), puppeteer: { headless: true, args: ['--no-sandbox', '--disable-setuid-sandbox'] } }); // Volver a registrar eventos client.on('qr', (qr) => { console.log('üì± Nuevo QR tras logout'); qrcode.generate(qr, { small: true }); currentQR = qr; }); client.on('ready', () => { console.log('ü§ñ Nuevo cliente listo tras logout'); currentQR = null; }); client.on('message', async (msg) => { // Igual que antes... const remitente = msg.from; const mensaje = msg.body; const fecha = new Date().toISOString(); db.run(` INSERT INTO mensajes (remitente, mensaje, fecha, fromMe) VALUES (?, ?, ?, 0) `, [remitente, mensaje, fecha]); await learningHandler.processMessage(msg); }); client.on('message_create', async (msg) => { if (msg.fromMe) { const remitente = msg.to; const mensaje = msg.body; const fecha = new Date().toISOString(); db.run(` INSERT INTO mensajes (remitente, mensaje, fecha, fromMe) VALUES (?, ?, ?, 1) `, [remitente, mensaje, fecha]); } }); await client.initialize(); console.log('‚úÖ Nuevo cliente inicializado tras logout.'); res.send(`

‚úÖ Sesi√≥n cerrada y cliente reiniciado

Puedes generar un nuevo QR para reconectar.

Volver al panel

`); } catch (err) { console.error('‚ùå Error en el cierre de sesi√≥n:', err); res.status(500).send('Error al cerrar sesi√≥n. Intente nuevamente.'); } }); module.exports = router;
